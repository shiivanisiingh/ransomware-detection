# Cost function with GitHub-friendly formatting
def calculate_cost(y_true: np.ndarray, y_pred: np.ndarray) -> float:
    """Implements 1:100 FP:FN cost ratio
    Args:
        y_true: Ground truth labels (0=legit, 1=ransomware)
        y_pred: Predicted labels
    Returns:
        Total operational cost in USD
    """
    fp = np.sum((y_pred == 1) & (y_true == 0))  # Type I error
    fn = np.sum((y_pred == 0) & (y_true == 1))  # Type II error
    return 100 * fp + 10000 * fn  # 1:100 ratio
